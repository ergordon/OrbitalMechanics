%% Part F

beta = 2*asin((sqrt((s-c)/(2*a))));

e = sqrt(1-(p/a));

%% Part G
gamma = asind((r2*sind(th))/c);
ui = [1;0];
uc = [-cosd(gamma); sind(gamma)];
A = sqrt(1/(4*a))*cot(alpha/2);

v1 = (B+A)*uc + (B-A)*ui;
aH = (r1+r2)/2;

end
